```sql
create database userdb1
on
(name=userdb4_data,--数据文件的逻辑名称,注意不能与日志逻辑同名
filename='g:\xygl\userdb4.mdf' ,--物理名称，注意路径必须存在
size=5,--数据初始长度为5M
maxsize=10,--最大长度为10M
filegrowth=1)--数据文件每次增长1M
log on
( name=userdb4_log,
filename='g:\xygl\userdb4.ldf ' ,
size=2 ,
maxsize=5 ,
filegrowth=1)
```
